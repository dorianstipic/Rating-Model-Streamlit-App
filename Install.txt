@echo off

REM Set the relative path to environment.txt in the same directory as the .bat file
set ENV_FILE=%~dp0environment.txt

REM Create the environment from the .txt file in the 'Environment' subfolder
call conda create --name camels_bat --file "%ENV_FILE%"

REM Check the exit code of the previous command
IF %ERRORLEVEL% EQU 0 (
    echo Environment 'camels_bat' has been installed successfully.
) ELSE (
    echo Error: Failed to install the environment 'camels_bat'.
)

REM Pause to keep the console window open
pause
